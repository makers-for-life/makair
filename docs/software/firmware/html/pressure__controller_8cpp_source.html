141  // Update the cycle phase
142  updatePhase(p_centiSec);
143 
144  if (!m_vigilance) {
145  // Act accordingly
146  switch (m_subPhase) {
147  case CycleSubPhases::INSPIRATION: {

446 int32_t PressureController::pidBlower(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
447  // Compute error
448  int32_t error = targetPressure - currentPressure;
449 
450  // Compute integral
451  blowerIntegral = blowerIntegral + ((PID_BLOWER_KI * error * dt) / 1000000);
452  blowerIntegral = max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, blowerIntegral));
453 
454  // Compute derivative
455  int32_t derivative = ((blowerLastError == INVALID_ERROR_MARKER) || (dt == 0))
456  ? 0
457  : ((1000000 * (error - blowerLastError)) / dt);
458  blowerLastError = error;
459 
460  int32_t blowerCommand = (PID_BLOWER_KP * error) + blowerIntegral
461  + ((PID_BLOWER_KD * derivative) / 1000); // Command computation
462 
463  int32_t minAperture = m_blower_valve.minAperture();
464  int32_t maxAperture = m_blower_valve.maxAperture();
465 
466  uint32_t blowerAperture =
467  max(minAperture,
468  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
469 
470  return blowerAperture;
471 }

473 int32_t
474 PressureController::pidPatient(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
475  // Compute error
476  int32_t error = targetPressure + PID_PATIENT_SAFETY_PEEP_OFFSET - currentPressure;
477 
478  // Compute integral
479  patientIntegral = patientIntegral + ((PID_PATIENT_KI * error * dt) / 1000000);
480  patientIntegral = max(PID_PATIENT_INTEGRAL_MIN, min(PID_PATIENT_INTEGRAL_MAX, patientIntegral));
481 
482  // Compute derivative
483  int32_t derivative = ((patientLastError == INVALID_ERROR_MARKER) || (dt == 0))
484  ? 0
485  : ((1000000 * (error - patientLastError)) / dt);
486  patientLastError = error;
487 
488  int32_t patientCommand = (PID_PATIENT_KP * error) + patientIntegral
489  + ((PID_PATIENT_KD * derivative) / 1000); // Command computation
490 
491  int32_t minAperture = m_blower_valve.minAperture();
492  int32_t maxAperture = m_blower_valve.maxAperture();
493 
494  uint32_t patientAperture =
495  max(minAperture,
496  min(maxAperture, maxAperture + (maxAperture - minAperture) * patientCommand / 1000));
497 
498  return patientAperture;
