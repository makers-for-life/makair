129 void PressureController::compute(uint16_t p_centiSec) {
130  // Update the cycle phase
131  updatePhase(p_centiSec);
132 
133  if (!m_vigilance) {
134  // Act accordingly
135  switch (m_subPhase) {
136  case CycleSubPhases::INSPIRATION: {
137  inhale();
138  break;
139  }
140  case CycleSubPhases::HOLD_INSPIRATION: {
141  plateau();
142  break;
143  }
144  case CycleSubPhases::EXHALE: {
145  exhale();
146  break;
147  }
148  case CycleSubPhases::HOLD_EXHALE: {
149  holdExhalation();
150  break;
151  }
152  default: {
153  holdExhalation();
154  }
155  }
156  }
157  safeguards(p_centiSec);
158 
159  DBG_PHASE_PRESSION(m_cycleNb, p_centiSec, 1u, m_phase, m_subPhase, m_pressure, m_blower.command,
160  m_blower.position, m_patient.command, m_patient.position)
161 
162  executeCommands();
163 }

250 void PressureController::updatePhase(uint16_t p_centiSec) {
251  if (p_centiSec < m_centiSecPerInhalation) {
252  m_phase = CyclePhases::INHALATION;
253 
254  if ((p_centiSec < ((m_centiSecPerInhalation * 80u) / 100u))

256  if (m_subPhase != CycleSubPhases::HOLD_INSPIRATION) {
257  m_consignePression = m_maxPeakPressureCommand;

392 int32_t PressureController::pidBlower(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
393  // Compute error
394  int32_t error = targetPressure - currentPressure;
395 
396  // Compute integral
397  blowerIntegral = blowerIntegral + ((PID_BLOWER_KI * error * dt) / 1000000);
398  blowerIntegral = max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, blowerIntegral));
399 
400  // Compute derivative
401  int32_t derivative = ((blowerLastError == INVALID_ERROR_MARKER) || (dt == 0))
402  ? 0
403  : ((1000000 * (error - blowerLastError)) / dt);
404  blowerLastError = error;
405 
406  int32_t blowerCommand = (PID_BLOWER_KP * error) + blowerIntegral
407  + ((PID_BLOWER_KD * derivative) / 1000); // Command computation
408 
409  int32_t minAperture = m_blower.minAperture();
410  int32_t maxAperture = m_blower.maxAperture();
411 
412  uint32_t blowerAperture =
413  max(minAperture,
414  min(maxAperture,
415  maxAperture - (blowerCommand + 1000) * (maxAperture - minAperture) / 2000));
416 
417  return blowerAperture;
418 }

420 int32_t
421 PressureController::pidPatient(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
422  // Compute error
423  // Increase target pressure by 20mm H2O for safety, to ensure from going below the target
424  // pressure
425  int32_t error = targetPressure + 20 - currentPressure;
426 
427  // Compute integral
428  patientIntegral = patientIntegral + ((PID_PATIENT_KI * error * dt) / 1000000);
429  patientIntegral = max(PID_PATIENT_INTEGRAL_MIN, min(PID_PATIENT_INTEGRAL_MAX, patientIntegral));
430 
431  // Compute derivative
432  int32_t derivative = ((patientLastError == INVALID_ERROR_MARKER) || (dt == 0))
433  ? 0
434  : ((1000000 * (error - patientLastError)) / dt);
435  patientLastError = error;
436 
437  int32_t patientCommand = (PID_PATIENT_KP * error) + patientIntegral
438  + ((PID_PATIENT_KD * derivative) / 1000); // Command computation
439 
440  int32_t minAperture = m_blower.minAperture();
441  int32_t maxAperture = m_blower.maxAperture();
442 
443  uint32_t patientAperture =
444  max(minAperture,
445  min(maxAperture, maxAperture + (maxAperture - minAperture) * patientCommand / 1000));
446 
447  return patientAperture;
448 }
